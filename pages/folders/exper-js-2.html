# 1.for-in循环注意事项

for-in循环不仅会遍历对象的实例属性，还会遍历从原型继承来的属性。出于这个原因，最好使用hasOwnProperty()方法来为for-in过滤出实例属性

let obj = {'name':'su','age':18,'sex':'male'};
for(var k in obj){
    if(obj.hasOwnProperty(k)){ //过滤
        console.log('属性key:'+k);
        console.log('属性值:'+obj[k]);
    }
}

# 3.设置localStorage有效期

Storage.prototype.setExpire = function (item, val, expire) {
    var obj = {
        val: val,
        tiem: Date.now(),
        expire: expire
    }
    localStorage.setItem(item, JSON.stringify(obj));
}

Storage.prototype.getExpire = function (item) {
    var localItem = JSON.parse(localStorage.getItem(item));
    if (!localItem) {
        return false;
    }
    if ((Date.now() - localItem.time) > localItem.expire) {
        localStorage.removeItem(item);
    }else{
        return localItem.val;
    }
}
